<script>
  (function () {
    const card = document.currentScript.previousElementSibling;
    const bars = card.querySelectorAll('.w-bars li');
    let max = 0;
    bars.forEach(li => max = Math.max(max, +li.dataset.value));
    bars.forEach(li => {
      const bar = li.querySelector('.w-bar');
      bar.style.width = (100 * (+li.dataset.value) / max) + '%';
      li.addEventListener('mouseenter', () => li.classList.add('w-hover'));
      li.addEventListener('mouseleave', () => li.classList.remove('w-hover'));
    });
  })();
</script>
