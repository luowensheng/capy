  <title>Breakout</title>
  <style>
    body { margin: 0; background: #0a0a14; display: grid; place-items: center; min-height: 100vh; font-family: ui-monospace, SFMono-Regular, monospace; color: #cfd; }
    canvas { background: #11111a; border: 2px solid #333; box-shadow: 0 0 40px #2bf3; }
    .wrap { text-align: center; }
    .hud  { margin: 8px 0; font-size: 13px; letter-spacing: 0.08em; }
    .hud span { color: #4ef; }
    .hint { color: #678; font-size: 11px; margin-top: 6px; }
  </style>
</head>
<body>
  <div class="wrap">
    <div class="hud">Breakout · score <span id="score">0</span> · lives <span id="lives">3</span></div>
    <canvas id="c" width="480" height="320"></canvas>
    <div class="hint">arrow keys to move · space to launch · R to restart</div>
  </div>
  <script>
    (() => {
      const W = 480, H = 320;
      const PW = 80, PH = 10, PSPD = 7;
      const PCOLOR = "#3df";
      const BR = 6, BSPD = 4;
      const BCOLOR = "#fff";
      const ROWS = 5, COLS = 8;
      const BW = 56, BH = 14, GAP = 4;
      const START_LIVES = 3;
      const ROW_COLORS = [ "#f55", "#fa4", "#ff4", "#4f6", "#4af",
      ];

      const c = document.getElementById("c"), ctx = c.getContext("2d");
      const scoreEl = document.getElementById("score"), livesEl = document.getElementById("lives");

      let paddle, ball, bricks, score, lives, launched, keys, gameOver;

      // -----------------------------------------------------------------
      // Action implementations. Names here MUST match the action idents
      // declared in the DSL via `on_key ... <action>` and `on_event ...
      // <action>`. Library authors add to this table when extending the DSL.
      // -----------------------------------------------------------------
      const ACTIONS = {
        paddle_left()       { paddle.x = Math.max(0, paddle.x - PSPD); },
        paddle_right()      { paddle.x = Math.min(W - PW, paddle.x + PSPD); },
        launch_ball()       {
          if (launched || gameOver) return;
          ball.vx = (Math.random() > 0.5 ? 1 : -1) * BSPD;
          ball.vy = -BSPD;
          launched = true;
        },
        reset()             { resetGame(); },

        destroy_brick(b)    { b.alive = false; },
        add_score(_b, value){
          score += value;
          scoreEl.textContent = score;
        },
        bounce_with_spin()  {
          ball.y = paddle.y - BR;
          ball.vy = -Math.abs(ball.vy);
          const off = (ball.x - (paddle.x + PW / 2)) / (PW / 2);
          ball.vx = off * (BSPD + 2);
        },
        lose_life()         {
          lives--;
          livesEl.textContent = lives;
          if (lives <= 0) { gameOver = "lose"; return; }
          ball.x = paddle.x + PW / 2;
          ball.y = paddle.y - BR - 1;
          ball.vx = 0; ball.vy = 0;
          launched = false;
        },
        win()               { gameOver = "win"; },
      };

      // -----------------------------------------------------------------
      // Generated bindings from the DSL.
      // -----------------------------------------------------------------
      const KEY_BINDINGS = {
        "ArrowLeft": "paddle_left",
        "ArrowRight": "paddle_right",
        " ": "launch_ball",
        "r": "reset",
      };

      const EVENT_HANDLERS = {
        brick_hit: (arg) => {
          ACTIONS["destroy_brick"] && ACTIONS["destroy_brick"](arg);
          ACTIONS["add_score"] && ACTIONS["add_score"](arg, 10);
        },
        paddle_hit: (arg) => {
          ACTIONS["bounce_with_spin"] && ACTIONS["bounce_with_spin"](arg);
        },
        ball_lost: (arg) => {
          ACTIONS["lose_life"] && ACTIONS["lose_life"](arg);
        },
        all_cleared: (arg) => {
          ACTIONS["win"] && ACTIONS["win"](arg);
        },
      };

      function emit(event, arg) {
        const h = EVENT_HANDLERS[event];
        if (h) h(arg);
      }

      function resetGame() {
        paddle = { x: (W - PW) / 2, y: H - PH - 8 };
        ball = { x: W / 2, y: paddle.y - BR - 1, vx: 0, vy: 0 };
        launched = false;
        bricks = [];
        const totalW = COLS * BW + (COLS - 1) * GAP;
        const offX = (W - totalW) / 2;
        const offY = 32;
        for (let r = 0; r < ROWS; r++) {
          for (let c = 0; c < COLS; c++) {
            bricks.push({
              x: offX + c * (BW + GAP),
              y: offY + r * (BH + GAP),
              w: BW, h: BH,
              color: ROW_COLORS[r % ROW_COLORS.length],
              alive: true,
              value: (ROWS - r) * 10,
            });
          }
        }
        score = 0;
        lives = START_LIVES;
        gameOver = null;
        scoreEl.textContent = score;
        livesEl.textContent = lives;
      }

      function step() {
        // Held-key actions: dispatch every frame for movement.
        for (const k in KEY_BINDINGS) {
          if (keys[k] && (k === "ArrowLeft" || k === "ArrowRight")) {
            ACTIONS[KEY_BINDINGS[k]] && ACTIONS[KEY_BINDINGS[k]]();
          }
        }

        if (!launched) {
          ball.x = paddle.x + PW / 2;
        } else {
          ball.x += ball.vx;
          ball.y += ball.vy;
        }

        if (ball.x - BR < 0)  { ball.x = BR; ball.vx = -ball.vx; }
        if (ball.x + BR > W)  { ball.x = W - BR; ball.vx = -ball.vx; }
        if (ball.y - BR < 0)  { ball.y = BR; ball.vy = -ball.vy; }
        if (ball.y - BR > H)  emit("ball_lost");

        if (
          launched &&
          ball.x > paddle.x && ball.x < paddle.x + PW &&
          ball.y + BR > paddle.y && ball.y + BR < paddle.y + PH + 8
        ) {
          emit("paddle_hit");
        }

        for (const b of bricks) {
          if (!b.alive) continue;
          if (
            ball.x + BR > b.x && ball.x - BR < b.x + b.w &&
            ball.y + BR > b.y && ball.y - BR < b.y + b.h
          ) {
            const fromLeft = ball.x < b.x, fromRight = ball.x > b.x + b.w;
            if (fromLeft || fromRight) ball.vx = -ball.vx;
            else ball.vy = -ball.vy;
            emit("brick_hit", b);
            break;
          }
        }

        if (bricks.every(b => !b.alive)) emit("all_cleared");
      }

      function draw() {
        ctx.clearRect(0, 0, W, H);
        for (const b of bricks) {
          if (!b.alive) continue;
          ctx.fillStyle = b.color;
          ctx.fillRect(b.x, b.y, b.w, b.h);
        }
        ctx.fillStyle = PCOLOR;
        ctx.fillRect(paddle.x, paddle.y, PW, PH);
        ctx.beginPath();
        ctx.arc(ball.x, ball.y, BR, 0, Math.PI * 2);
        ctx.fillStyle = BCOLOR;
        ctx.fill();
        if (gameOver) {
          ctx.fillStyle = "rgba(10, 10, 20, 0.7)";
          ctx.fillRect(0, 0, W, H);
          ctx.fillStyle = "#cfd";
          ctx.font = "bold 22px ui-monospace, monospace";
          ctx.textAlign = "center";
          ctx.fillText(gameOver === "win" ? "YOU WIN" : "GAME OVER", W / 2, H / 2 - 6);
          ctx.font = "13px ui-monospace, monospace";
          ctx.fillText("press R to restart", W / 2, H / 2 + 18);
        }
      }

      function loop() { if (!gameOver) step(); draw(); requestAnimationFrame(loop); }

      keys = {};
      document.addEventListener("keydown", (e) => {
        keys[e.key] = true;
        const action = KEY_BINDINGS[e.key];
        // One-shot actions fire on keydown; held-direction actions are
        // dispatched every frame in step(). Discriminate by key.
        if (action && e.key !== "ArrowLeft" && e.key !== "ArrowRight") {
          ACTIONS[action] && ACTIONS[action]();
          e.preventDefault();
        }
      });
      document.addEventListener("keyup", (e) => { keys[e.key] = false; });

      resetGame();
      loop();
    })();
  </script>
