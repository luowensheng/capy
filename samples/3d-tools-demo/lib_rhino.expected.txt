using System.Drawing;
using Rhino;
using Rhino.DocObjects;
using Rhino.Geometry;
using Rhino.Geometry.Light;

public static class CapyScene
{
    static readonly System.Collections.Generic.Dictionary<string, Color> COLORS =
        new System.Collections.Generic.Dictionary<string, Color> {
            { "red", Color.FromArgb(220, 40, 40) },
            { "blue", Color.FromArgb(40, 80, 220) },
            { "green", Color.FromArgb(40, 200, 80) },
            { "gray", Color.FromArgb(150, 150, 150) },
        };

    static ObjectAttributes Attr(string color)
    {
        var a = new ObjectAttributes { ColorSource = ObjectColorSource.ColorFromObject };
        a.ObjectColor = COLORS.TryGetValue(color, out var c) ? c : Color.White;
        return a;
    }

    public static void Build(RhinoDoc doc)
    {
        // scene: "Studio"
        // cube (red)
        {
            var c = new Point3d(0, 0, 0);
            double s = 2 / 2.0;
            var box = new Box(new Plane(c, Vector3d.ZAxis), new Interval(-s, s), new Interval(-s, s), new Interval(-s, s));
            doc.Objects.AddBox(box, Attr("red"));
        }
        // sphere (blue)
        {
            var sph = new Sphere(new Point3d(4, 0, 0), 1);
            doc.Objects.AddSphere(sph, Attr("blue"));
        }
        // plane (gray)
        {
            var c = new Point3d(0, 0, 0);
            double s = 10 / 2.0;
            var rect = new Rectangle3d(new Plane(c, Vector3d.ZAxis), new Interval(-s, s), new Interval(-s, s));
            doc.Objects.AddRectangle(rect, Attr("gray"));
        }
        // point light
        {
            var lt = new Light { LightStyle = LightStyle.WorldPoint, Location = new Point3d(5, 5, 5) };
            doc.Lights.Add(lt);
        }
        // camera
        doc.Views.ActiveView.ActiveViewport.SetCameraLocation(new Point3d(7, 7, 3), true);
        doc.Views.ActiveView.ActiveViewport.SetCameraTarget(new Point3d(0, 0, 0), true);

    }
}
