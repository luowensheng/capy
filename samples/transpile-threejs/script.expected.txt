    <small>drag&nbsp;= orbit · scroll&nbsp;= zoom · click an object · keys: see HUD</small>

    <button data-target="hero" data-action="randomize_color">🎨 Randomize hero</button>
    <button data-target="beacon" data-action="explode">💥 Explode beacon</button>

  <script type="importmap">
  {
    "imports": {
      "three": "https://unpkg.com/three@0.160.0/build/three.module.js",
      "three/addons/": "https://unpkg.com/three@0.160.0/examples/jsm/"
    }
  }
  </script>
  <script type="module">
    import * as THREE from "three";
    import { OrbitControls } from "three/addons/controls/OrbitControls.js";

    const scene = new THREE.Scene();
    scene.background = new THREE.Color("#0a0a14");

    const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 100);
    camera.position.set(0, 1.5, 7);

    const renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setPixelRatio(window.devicePixelRatio);
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.body.appendChild(renderer.domElement);

    const controls = new OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;

    // Ambient light
    scene.add(new THREE.AmbientLight("#ffffff", 0.45));

    // Directional lights
    {
      const L = new THREE.DirectionalLight("#ffffff", 0.9);
      L.position.set(5, 6, 3);
      scene.add(L);
    }
    {
      const L = new THREE.DirectionalLight("#88aaff", 0.4);
      L.position.set(-4, 2, -3);
      scene.add(L);
    }

    // Meshes
    const meshes = {};
    {
      let geo;
      geo = new THREE.BoxGeometry(1, 1, 1);
      const mat = new THREE.MeshStandardMaterial({ color: "#4dd6c1", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(0, 0, 0);
      m.userData = {
        name: "hero",
        motion: "spin",
        defaultMotion: "spin",
        color0: "#4dd6c1",
        t0: { x: 0, y: 0, z: 0 },
        paused: false,
        wireframe: false,
        explodeT: 0,
      };
      scene.add(m);
      meshes["hero"] = m;
    }
    {
      let geo;
      geo = new THREE.SphereGeometry(0.6, 32, 32);
      const mat = new THREE.MeshStandardMaterial({ color: "#f6a05a", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(-2.5, 1, 0);
      m.userData = {
        name: "planet",
        motion: "orbit",
        defaultMotion: "orbit",
        color0: "#f6a05a",
        t0: { x: -2.5, y: 1, z: 0 },
        paused: false,
        wireframe: false,
        explodeT: 0,
      };
      scene.add(m);
      meshes["planet"] = m;
    }
    {
      let geo;
      geo = new THREE.TorusGeometry(0.55, 0.18, 16, 64);
      const mat = new THREE.MeshStandardMaterial({ color: "#fdd835", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(2.5, -0.6, 0);
      m.userData = {
        name: "ring",
        motion: "spin",
        defaultMotion: "spin",
        color0: "#fdd835",
        t0: { x: 2.5, y: -0.6, z: 0 },
        paused: false,
        wireframe: false,
        explodeT: 0,
      };
      scene.add(m);
      meshes["ring"] = m;
    }
    {
      let geo;
      geo = new THREE.ConeGeometry(0.55, 1.0, 32);
      const mat = new THREE.MeshStandardMaterial({ color: "#b388ff", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(0, 1.8, -2);
      m.userData = {
        name: "beacon",
        motion: "bob",
        defaultMotion: "bob",
        color0: "#b388ff",
        t0: { x: 0, y: 1.8, z: -2 },
        paused: false,
        wireframe: false,
        explodeT: 0,
      };
      scene.add(m);
      meshes["beacon"] = m;
    }
    {
      let geo;
      geo = new THREE.PlaneGeometry(4, 4);
      const mat = new THREE.MeshStandardMaterial({ color: "#1f1f2e", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(0, -1.2, 0);
      m.rotation.x = -Math.PI / 2;
      m.userData = {
        name: "floor",
        motion: "none",
        defaultMotion: "none",
        color0: "#1f1f2e",
        t0: { x: 0, y: -1.2, z: 0 },
        paused: false,
        wireframe: false,
        explodeT: 0,
      };
      scene.add(m);
      meshes["floor"] = m;
    }

    // ── Status + action dispatch ────────────────────────────────
    const status = document.getElementById("status");
    let statusTimer = 0;
    function setStatus(s) {
      status.textContent = s;
      clearTimeout(statusTimer);
      statusTimer = setTimeout(() => { status.textContent = "ready"; }, 1800);
    }

    let disco = false;
    const motionCycle = ["spin", "orbit", "bob", "pulse", "none"];

    function applyAction(target, action) {
      const list = (target === "any" || target === "all")
        ? Object.values(meshes)
        : (meshes[target] ? [meshes[target]] : []);
      if (list.length === 0) { setStatus("no such mesh: " + target); return; }

      for (const m of list) {
        switch (action) {
          case "randomize_color":
            m.material.color.setHSL(Math.random(), 0.6, 0.55);
            break;
          case "toggle_motion":
            m.userData.paused = !m.userData.paused;
            break;
          case "cycle_motion": {
            const i = motionCycle.indexOf(m.userData.motion);
            m.userData.motion = motionCycle[(i + 1) % motionCycle.length];
            break;
          }
          case "reset":
            m.position.set(m.userData.t0.x, m.userData.t0.y, m.userData.t0.z);
            m.rotation.set(0, 0, 0);
            m.scale.set(1, 1, 1);
            m.material.color.set(m.userData.color0);
            m.material.wireframe = false;
            m.userData.paused = false;
            m.userData.motion = m.userData.defaultMotion;
            m.userData.explodeT = 0;
            break;
          case "reset_all":
            for (const k in meshes) applyAction(k, "reset");
            disco = false;
            return;
          case "explode":
            m.userData.explodeT = 1.0;
            break;
          case "wireframe":
            m.userData.wireframe = !m.userData.wireframe;
            m.material.wireframe = m.userData.wireframe;
            break;
          case "disco":
            disco = !disco;
            setStatus(disco ? "disco ON" : "disco OFF");
            return;
          default:
            setStatus("unknown action: " + action);
            return;
        }
      }
      setStatus(action + " → " + target);
    }

    // ── Raycaster: click + hover dispatch ───────────────────────
    const raycaster = new THREE.Raycaster();
    const pointer = new THREE.Vector2();
    function pickMesh(ev) {
      const rect = renderer.domElement.getBoundingClientRect();
      pointer.x = ((ev.clientX - rect.left) / rect.width) * 2 - 1;
      pointer.y = -((ev.clientY - rect.top) / rect.height) * 2 + 1;
      raycaster.setFromCamera(pointer, camera);
      const hits = raycaster.intersectObjects(Object.values(meshes), false);
      return hits.length ? hits[0].object : null;
    }

    const clickBindings = {
      "hero": "cycle_motion",
      "planet": "randomize_color",
      "ring": "wireframe",
      "beacon": "explode",
    };
    const hoverBindings = {
      "hero": "randomize_color",
    };

    renderer.domElement.addEventListener("click", (ev) => {
      const obj = pickMesh(ev);
      if (!obj) return;
      const action = clickBindings[obj.userData.name];
      if (action) applyAction(obj.userData.name, action);
    });

    let lastHover = null;
    renderer.domElement.addEventListener("pointermove", (ev) => {
      const obj = pickMesh(ev);
      renderer.domElement.style.cursor = obj && clickBindings[obj.userData.name] ? "pointer" : "grab";
      if (obj !== lastHover) {
        if (obj && hoverBindings[obj.userData.name]) {
          applyAction(obj.userData.name, hoverBindings[obj.userData.name]);
        }
        lastHover = obj;
      }
    });

    // ── Keyboard bindings ───────────────────────────────────────
    const keyBindings = [
      { key: " ", target: "hero", action: "toggle_motion" },
      { key: "r", target: "any", action: "reset_all" },
      { key: "x", target: "planet", action: "explode" },
      { key: "d", target: "any", action: "disco" },
      { key: "w", target: "ring", action: "wireframe" },
    ];
    window.addEventListener("keydown", (e) => {
      for (const b of keyBindings) {
        if (e.key === b.key) {
          applyAction(b.target, b.action);
          e.preventDefault();
          return;
        }
      }
    });

    // ── HUD buttons ─────────────────────────────────────────────
    for (const btn of document.querySelectorAll("#hud button")) {
      btn.addEventListener("click", () => {
        applyAction(btn.dataset.target, btn.dataset.action);
      });
    }

    window.addEventListener("resize", () => {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    });

    const clock = new THREE.Clock();
    let discoT = 0;
    function tick() {
      const t = clock.getElapsedTime();
      const dt = Math.min(0.05, clock.getDelta());

      if (disco) {
        discoT += dt;
        if (discoT > 0.18) {
          discoT = 0;
          for (const k in meshes) {
            meshes[k].material.color.setHSL(Math.random(), 0.7, 0.55);
          }
        }
      }

      for (const name in meshes) {
        const m = meshes[name];
        if (m.userData.paused) continue;
        const motion = m.userData.motion;
        const t0 = m.userData.t0;
        if (motion === "spin") {
          m.rotation.x += 0.01;
          m.rotation.y += 0.015;
        } else if (motion === "orbit") {
          const r = Math.hypot(t0.x, t0.z) || 2;
          const phase = Math.atan2(t0.z, t0.x);
          m.position.x = r * Math.cos(t * 0.6 + phase);
          m.position.z = r * Math.sin(t * 0.6 + phase);
          m.rotation.y += 0.02;
        } else if (motion === "bob") {
          m.position.y = t0.y + Math.sin(t * 2) * 0.3;
        } else if (motion === "pulse") {
          const s = 1 + Math.sin(t * 3) * 0.15;
          m.scale.set(s, s, s);
        }

        // Explode animation — independent of motion, decays back
        // to identity scale. Triggered by `explode` action.
        if (m.userData.explodeT > 0) {
          m.userData.explodeT = Math.max(0, m.userData.explodeT - dt * 1.6);
          const k = m.userData.explodeT;
          const s = 1 + k * 1.2;
          m.scale.multiplyScalar(s / (m.scale.x || 1));
          m.material.opacity = 1 - k * 0.4;
          m.material.transparent = k > 0.01;
          if (m.userData.explodeT === 0) {
            m.scale.set(1, 1, 1);
            m.material.opacity = 1;
            m.material.transparent = false;
          }
        }
      }
      controls.update();
      renderer.render(scene, camera);
      requestAnimationFrame(tick);
    }
    tick();
  </script>
