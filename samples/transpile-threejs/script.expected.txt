  <style>
    html, body { margin: 0; height: 100%; background: #0a0a14; color: #eee; font-family: system-ui, sans-serif; overflow: hidden; }
    #title { position: fixed; top: 10px; left: 12px; font-size: 13px; opacity: 0.75; pointer-events: none; }
    canvas { display: block; }
  </style>

<body>
  <div id="title">Capy ✕ three.js — drag to orbit</div>
  <script type="importmap">
  {
    "imports": {
      "three": "https://unpkg.com/three@0.160.0/build/three.module.js",
      "three/addons/": "https://unpkg.com/three@0.160.0/examples/jsm/"
    }
  }
  </script>
  <script type="module">
    import * as THREE from "three";
    import { OrbitControls } from "three/addons/controls/OrbitControls.js";

    const scene = new THREE.Scene();
    scene.background = new THREE.Color("#0a0a14");

    const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 100);
    camera.position.set(0, 1.5, 7);

    const renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setPixelRatio(window.devicePixelRatio);
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.body.appendChild(renderer.domElement);

    const controls = new OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;

    // Ambient light
    scene.add(new THREE.AmbientLight("#ffffff", 0.45));

    // Directional lights
    {
      const L = new THREE.DirectionalLight("#ffffff", 0.9);
      L.position.set(5, 6, 3);
      scene.add(L);
    }
    {
      const L = new THREE.DirectionalLight("#88aaff", 0.4);
      L.position.set(-4, 2, -3);
      scene.add(L);
    }

    // Meshes
    const meshes = {};
    {
      let geo;
      geo = new THREE.BoxGeometry(1, 1, 1);
      const mat = new THREE.MeshStandardMaterial({ color: "#4dd6c1", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(0, 0, 0);
      m.userData = { motion: "spin", t0: { x: 0, y: 0, z: 0 } };
      scene.add(m);
      meshes["hero"] = m;
    }
    {
      let geo;
      geo = new THREE.SphereGeometry(0.6, 32, 32);
      const mat = new THREE.MeshStandardMaterial({ color: "#f6a05a", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(-2.5, 1, 0);
      m.userData = { motion: "orbit", t0: { x: -2.5, y: 1, z: 0 } };
      scene.add(m);
      meshes["planet"] = m;
    }
    {
      let geo;
      geo = new THREE.TorusGeometry(0.55, 0.18, 16, 64);
      const mat = new THREE.MeshStandardMaterial({ color: "#fdd835", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(2.5, -0.6, 0);
      m.userData = { motion: "spin", t0: { x: 2.5, y: -0.6, z: 0 } };
      scene.add(m);
      meshes["ring"] = m;
    }
    {
      let geo;
      geo = new THREE.ConeGeometry(0.55, 1.0, 32);
      const mat = new THREE.MeshStandardMaterial({ color: "#b388ff", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(0, 1.8, -2);
      m.userData = { motion: "bob", t0: { x: 0, y: 1.8, z: -2 } };
      scene.add(m);
      meshes["beacon"] = m;
    }
    {
      let geo;
      geo = new THREE.PlaneGeometry(4, 4);
      const mat = new THREE.MeshStandardMaterial({ color: "#1f1f2e", roughness: 0.5, metalness: 0.1 });
      const m = new THREE.Mesh(geo, mat);
      m.position.set(0, -1.2, 0);
      m.rotation.x = -Math.PI / 2;
      m.userData = { motion: "none", t0: { x: 0, y: -1.2, z: 0 } };
      scene.add(m);
      meshes["floor"] = m;
    }

    window.addEventListener("resize", () => {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    });

    const clock = new THREE.Clock();
    function tick() {
      const t = clock.getElapsedTime();
      for (const name in meshes) {
        const m = meshes[name];
        const motion = m.userData.motion;
        const t0 = m.userData.t0;
        if (motion === "spin") {
          m.rotation.x += 0.01;
          m.rotation.y += 0.015;
        } else if (motion === "orbit") {
          const r = Math.hypot(t0.x, t0.z) || 2;
          const phase = Math.atan2(t0.z, t0.x);
          m.position.x = r * Math.cos(t * 0.6 + phase);
          m.position.z = r * Math.sin(t * 0.6 + phase);
          m.rotation.y += 0.02;
        } else if (motion === "bob") {
          m.position.y = t0.y + Math.sin(t * 2) * 0.3;
        } else if (motion === "pulse") {
          const s = 1 + Math.sin(t * 3) * 0.15;
          m.scale.set(s, s, s);
        }
      }
      controls.update();
      renderer.render(scene, camera);
      requestAnimationFrame(tick);
    }
    tick();
  </script>
